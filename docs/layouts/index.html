Quick Installation

# 1. Clone the repository
git clone https://github.com/tranqy/zellij-utils.git
cd zellij-utils

# 2. Run the installer
chmod +x scripts/install.sh
./scripts/install.sh

# 3. Add to your shell configuration
echo 'source ~/.config/shell/zellij-utils.sh' >> ~/.bashrc
source ~/.bashrc
